$$
v_K^+(t_2) - v_K^+(t_1) = \int_{s \in (t_1, t_2]} v_s \beta_K(ds)
$$
$$
v_K^+(t) - v_K^-(t) = v_t \beta_K\left( \left\{ t \right\} \right)
